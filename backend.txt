function hashPin(pin) {
  const raw = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, pin, Utilities.Charset.UTF_8);
  return raw.map(b => (b < 0 ? b + 256 : b).toString(16).padStart(2, '0')).join("");
}

/**
 * Normalise le texte en un "slug" (minuscules, sans accents, sans caractères spéciaux)
 * Cela permet de comparer "La Préparation d'un Ministre" et "preparation_ministre" équitablement.
 */
function toSlug(text) {
  if (!text) return "";
  return String(text)
    .toLowerCase()
    .normalize("NFD")               // Décompose les caractères accentués
    .replace(/[\u0300-\u036f]/g, "") // Supprime les accents
    .replace(/[^a-z0-9]/g, "_")      // Remplace tout ce qui n'est pas alphanumérique par "_"
    .replace(/_+/g, "_")             // Remplace les doubles underscores par un seul
    .replace(/^_|_$/g, "");          // Supprime les underscores au début et à la fin
}

/**
 * UNIVERSAL HEADER FINDER
 * Finds the index of the email/username column regardless of Case, Spaces, or variations.
 */
function findEmailColumn(headers) {
  // Define all acceptable column names here
  const candidates = ["username", "email", "student email", "email address", "user"];
  
  const normalizedHeaders = headers.map(h => String(h).toLowerCase().trim());
  
  for (let c of candidates) {
    const idx = normalizedHeaders.indexOf(c);
    if (idx !== -1) return idx;
  }
  return -1;
}

    }

    log("Scanning column index " + loginColIdx + " for user: " + identity);
    const row = data.slice(1).find(r => String(r[loginColIdx]).toLowerCase().trim() === identity.toLowerCase());

    if (!row) {
      log("FATAL: User '" + identity + "' not found in row data.");
      return { success: false, error: "AUTH_FAIL: Utilisateur introuvable.", logs: DEBUG_LOGS.join("\n") };
    }

      const idx = headersNorm.indexOf(colName.toLowerCase());
      return idx !== -1 && String(row[idx]).toUpperCase() === "TRUE";
    };

    const roleIdx = headersNorm.indexOf("role");

    return {
      success: true, // Mark as success
      email: identity,
      role: roleIdx !== -1 ? row[roleIdx] : "Staff",
      permissions: {
        p_pic: checkPerm("p_pic"),
        p_name: checkPerm("p_name"),

        p_settings: checkPerm("p_settings"),
        p_manual: checkPerm("p_manual"),
        p_review: checkPerm("p_review"),
        p_build: checkPerm("p_build")

  } catch (e) {
    // CATCH ALL CRASHES AND RETURN SAFELY
    log("❌ CRITICAL EXCEPTION: " + e.toString());
    return { 
      success: false, 
      error: "SERVER_CRASH: " + e.message,

/**
 * Mise à jour pour inclure le ReleaseMode et le Slug pour le lien de consultation
 */
function getSpecificExamSettings(examTitle, studentCode = null) {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  const settingsSh = ss.getSheetByName(SETTINGS_SHEET);
  const data = settingsSh.getDataRange().getValues();
  const headers = data[0];
  
  const titleIdx = headers.indexOf("ExamTitle");
  const acceptIdx = headers.indexOf("AcceptResponse") !== -1 ? headers.indexOf("AcceptResponse") : headers.indexOf("AcceptResponses");
  const releaseIdx = headers.indexOf("ReleaseMode");

  const row = data.find(r => String(r[titleIdx]).trim() === String(examTitle).trim());
  if (!row) return { AcceptResponses: "FALSE" };

  const releaseMode = String(row[releaseIdx] || "hidden").toLowerCase();
  const slug = toSlug(examTitle);

  // Vérification si déjà réussi
  if (studentCode) {
    const scoresSh = ss.getSheetByName(SCORES_SHEET);
    const sData = scoresSh.getDataRange().getValues();
    const sHeaders = sData[0];
    const examColIdx = sHeaders.indexOf(examTitle);
    const studentRow = sData.find(r => String(r[0]).trim() === String(studentCode).trim());

    if (studentRow && examColIdx !== -1) {
      const score = parseFloat(studentRow[examColIdx]);
      if (!isNaN(score) && score >= 70) {
        return { 
          AcceptResponses: "ALREADY_PASSED", 
          score: score, 
          releaseMode: releaseMode,
          examSlug: slug 
        };
      }
    }
  }

  return { 
    AcceptResponses: String(row[acceptIdx]).toUpperCase(),
    releaseMode: releaseMode,
    examSlug: slug
  };
}

/**
 * Enregistre une note manuellement et envoie une notification par email.
 * @param {Object} obj Contient studentCode, examSlug, et grade.
 * @return {Object} Résultat avec succès (boolean) et message pour l'interface.
 */
function saveManualExamGrade(obj) {
  try {
    const { studentCode, examSlug, grade } = obj;

    // Validation de base des entrées
    if (!studentCode || !examSlug || grade === "") {
      return { success: false, message: "Données manquantes (Code, Sujet ou Note)." };
    }

    const ss = SpreadsheetApp.openById(SUBMISSION_SHEET_ID);
    const sh = ss.getSheetByName(SCORES_SHEET);

    if (!sh) {
      throw new Error("La feuille '" + SCORES_SHEET + "' est introuvable.");
    }

    const data = sh.getDataRange().getValues();
    const headers = data[0].map(h => String(h).trim());

    const codeIdx = headers.indexOf("Student Code");
    const examTitle = String(examSlug).trim();
    const examColIdx = headers.indexOf(examTitle);

    if (codeIdx === -1) {
      throw new Error("Colonne 'Student Code' introuvable dans Scores.");
    }
    if (examColIdx === -1) {
      throw new Error("Matière introuvable dans les colonnes : " + examTitle);
    }

    // Trouver la ligne de l'étudiant
    let rowIndex = -1;
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][codeIdx]).trim() === String(studentCode).trim()) {
        rowIndex = i + 1; // +1 car index commence à 0
        break;
      }
    }

    // Si l'étudiant n'existe pas encore dans la feuille Scores, on l'ajoute
    if (rowIndex === -1) {
      sh.appendRow([studentCode]);
      rowIndex = sh.getLastRow();
    }

    // 1️⃣ Enregistrement de la note
    sh.getRange(rowIndex, examColIdx + 1).setValue(grade);

    // 2️⃣ Récupération des infos de l'étudiant pour l'email
    const userSh = ss.getSheetByName(MASTER_SHEET_NAME);
    let studentEmail = "";
    let studentName = studentCode;

    if (userSh) {
      const userData = userSh.getDataRange().getValues();
      const userHeaders = userData[0].map(h => String(h).trim());

      const idIdx = userHeaders.indexOf("Student ID");
      const nameIdx = userHeaders.indexOf("Student_Name");
      let emailIdx = userHeaders.indexOf("Email");
      if (emailIdx === -1) emailIdx = userHeaders.indexOf("Student Email");

      for (let i = 1; i < userData.length; i++) {
        if (String(userData[i][idIdx]).trim() === String(studentCode).trim()) {
          studentEmail = emailIdx !== -1 ? String(userData[i][emailIdx]).trim() : "";
          studentName = nameIdx !== -1 ? String(userData[i][nameIdx]).trim() : studentName;
          break;
        }
      }
    }

    // 3️⃣ Envoi de l'email avec validation rigoureuse
    // Regex pour valider le format de l'email et éviter l'erreur "Invalid email"
    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    
    if (studentEmail && emailPattern.test(studentEmail)) {
      try {
        MailApp.sendEmail({
          to: studentEmail,
          subject: `Note publiée — ${examTitle}`,
          htmlBody: `
            <div style="font-family: Arial, sans-serif; color: #333;">
              <p>Bonjour <b>${studentName}</b>,</p>
              <p>Votre note pour la matière <b>${examTitle}</b> a été enregistrée avec succès.</p>
              <div style="background: #f4f4f8; padding: 15px; border-radius: 8px; display: inline-block; border: 1px solid #ddd;">
                <span style="font-size: 18px;">Score : <b>${grade} / 100</b></span>
              </div>
              <p>Vous pouvez consulter les détails sur votre portail étudiant.</p>
              <hr style="border:none; border-top:1px solid #eee; margin: 20px 0;">
              <p style="font-size: 11px; color: #777;">Ceci est un message automatique généré par le système Living Word.</p>
            </div>
          `
        });
      } catch (mailErr) {
        Logger.log("Erreur d'envoi mail (mais note enregistrée) : " + mailErr.message);
        return { success: true, message: "Note enregistrée, mais l'email n'a pas pu être envoyé (Vérifiez l'adresse)." };
      }
    } else if (studentEmail !== "") {
       return { success: true, message: "Note enregistrée. Attention : l'email '" + studentEmail + "' est invalide." };
    }

    return { success: true, message: "Note enregistrée et notification envoyée !" };

  } catch (e) {
    Logger.log("ERREUR CRITIQUE saveManualExamGrade : " + e.message);
    return { success: false, message: "Erreur système : " + e.message };
  }
}
function updateUserRoleAndPerms(obj) {
  try {
    const ss = SpreadsheetApp.openById(SUBMISSION_SHEET_ID);
    const sh = ss.getSheetByName("Users");
    const fullData = sh.getDataRange().getValues();
    const headers = fullData[0].map(h => String(h).trim());

    // --- ÉTAPE A : GÉRER LA COLONNE p_name SI MANQUANTE ---
    let nameColIdx = headers.indexOf("p_name");
    if (nameColIdx === -1) {
      nameColIdx = headers.length; // Index de la nouvelle colonne
      sh.getRange(1, nameColIdx + 1).setValue("p_name").setFontWeight("bold");
      headers.push("p_name"); // Mise à jour des en-têtes en mémoire
    }

    // --- ÉTAPE B : TROUVER OU CRÉER L'UTILISATEUR ---
    let rowIndex = -1;
    for (let i = 1; i < fullData.length; i++) {
      if (String(fullData[i][0]).toLowerCase() === obj.email.toLowerCase()) {
        rowIndex = i + 1;
        break;
      }
    }

    // Si c'est une création (pas trouvé), on ajoute une nouvelle ligne
    if (rowIndex === -1) {
      // On prépare une ligne vide de la bonne longueur
      let newRow = new Array(headers.length).fill("");
      newRow[0] = obj.email;
      sh.appendRow(newRow);
      rowIndex = sh.getLastRow();
    }

    // --- ÉTAPE C : ENREGISTRER LES PERMISSIONS ---
    // On utilise les index dynamiques pour éviter les décalages de colonnes
    const mapping = {
      "Role": obj.role,
      "p_dossier": obj.p_dossier,
      "p_settings": obj.p_settings,
      "p_manual": obj.p_manual,
      "p_review": obj.p_review,
      "p_build": obj.p_build,
      "p_pic": obj.p_pic,
      "p_name": obj.p_name // La nouvelle permission
    };

    // Si un mot de passe est fourni, on le met à jour
    if (obj.password) {
      sh.getRange(rowIndex, headers.indexOf("Password") + 1).setValue(obj.password);
    }

    // On boucle sur le mapping pour écrire dans les bonnes colonnes
    for (let key in mapping) {
      let colIdx = headers.indexOf(key);
      if (colIdx !== -1) {
        sh.getRange(rowIndex, colIdx + 1).setValue(mapping[key]);
      }
    }

    return { success: true, status: rowIndex === -1 ? "registered" : "updated" };

  } catch (e) {
    return { success: false, message: e.toString() };
  }
}
function isAdminEmail(email) {
  if (!email) return false;
  email = email.toLowerCase().trim();
  
  // 1. Hardcoded Safety Net (Always allows you in)
  const masterAdmins = ["ministryschoolpap@gmail.com", "meigensmetellus@gmail.com"];
  if (masterAdmins.includes(email)) return true;

  // 2. Sheet Lookup
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  let sh = ss.getSheetByName("Users");
  if (!sh) return false;

  const data = sh.getDataRange().getValues();
  const headers = data[0];
  
  // Use the robust finder
  const emailIdx = findEmailColumn(headers);

  if (emailIdx === -1) {
    Logger.log("CRITICAL: No Email/Username column found in Users sheet.");
    return false;
  }

  // 3. Scan rows (Trimmed and Lowercase comparison)
  const found = data.slice(1).some(row => String(row[emailIdx]).toLowerCase().trim() === email);
  
  if (!found) Logger.log("Admin check failed: " + email + " not found in DB.");
  return found;
}
/*************************************************
 * getAvailableExams()
 * - reads Scores header and returns an array of {slug, title}
 * - header columns from 3rd column onward (0-based) are considered exam columns
 * - slug will be header value but normalized to a slug
 *************************************************/
function getAvailableExams() {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  let sh = ss.getSheetByName(SCORES_SHEET);
  if (!sh) return [];
  const header = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0].map(h => String(h||"").trim());
  // Standard: Student Code, Student Name, Picture_URL in first 3 columns
  const exams = [];
  for (let i = 3; i < header.length; i++) {
    const title = header[i];
    if (!title) continue;
    // slug: remove spaces, lowercase, replace non-alphanum with underscore
    const slug = title.toString().toLowerCase().replace(/\s+/g,'_').replace(/[^a-z0-9_]/g,'');
    exams.push({ slug: slug, title: title });
  }
  // If no exams found, fall back to common slugs
  if (exams.length === 0) {
    exams.push({ slug: "nt", title: "Enquête du Nouveau Testament" });
  }
  return exams;
}

/*************************************************
 * getAllSubmissions() - admin-only
 * Reads Scores sheet and returns structured objects
 *************************************************/
function getAllSubmissions() {
  const email = Session.getActiveUser().getEmail();
  if (!isAdminEmail(email)) throw new Error("Access denied");

  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  let sh = ss.getSheetByName(SCORES_SHEET);
  if (!sh) return [];

  const values = sh.getDataRange().getValues();
  if (!values || values.length === 0) return [];

  const header = values[0].map(c => String(c||"").trim());
  const rows = [];
  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    const obj = {};
    header.forEach((h, j) => {
      obj[h] = row[j] !== undefined ? row[j] : "";
    });
    obj._row = r+1;
    // try parse Submission_JSON if present
    if (obj["Submission_JSON"]) {
      try { obj.Submission_JSON_parsed = JSON.parse(obj["Submission_JSON"]); } catch(e) { obj.Submission_JSON_parsed = null; }
    } else {
      obj.Submission_JSON_parsed = null;
    }

// Remove the old setLoggedInStudent function entirely. It is dangerous.

function getLoggedInStudent(token) {
  // If no token is provided, stop immediately.
  if (!token) return { success: false };

  // Fetch the data associated with the token
  const cachedData = CacheService.getScriptCache().get(token);
  
  // If the token is expired or invalid, stop.
  if (!cachedData) return { success: false };

  // --- THE FIX: TRY / CATCH BLOCK ---
  try {
    // We try to read it as Student Data (JSON)
    const student = JSON.parse(cachedData);
    
    // If it works and has an ID, it's a valid Student
    if (student && student.Student_ID) {
      return { success: true, student: student };
    }
  } catch (e) {
    // 🚨 ERROR CAUGHT! 
    // This happens when an ADMIN logs in. Their data is just text (e.g. "email@gmail.com").
    // JSON.parse fails on plain text.
    // We catch the error silently and return false. 
    // This lets doGet() continue running so it can load the Admin Panel.
    return { success: false };
  }

  return { success: false };
}
/**
 * Charge le rapport d'examen en vérifiant le statut de libération dans Settings
 */
function loadStudentExam(slug) {
  const logged = getLoggedInStudent(); // Vérifie la session
  if (!logged.success) return null;

  const studentCode = String(logged.student.Student_ID).trim();
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID); //
  
  // 1. Détecter le titre de l'examen via le slug
  const scoresSheet = ss.getSheetByName(SCORES_SHEET); //
  const header = scoresSheet.getDataRange().getValues()[0].map(h => String(h || "").trim());
  
  const examTitle = header.find(t => t.toLowerCase().replace(/\s+/g, "") === slug.toLowerCase().replace(/\s+/g, ""));
  if (!examTitle) return null;

  // 2. Vérifier le ReleaseMode dans la feuille Settings
  const settingsSh = ss.getSheetByName(SETTINGS_SHEET); //
  const sData = settingsSh.getDataRange().getValues();
  const sHeaders = sData[0];
  const titleIdx = sHeaders.indexOf("ExamTitle"); //
  const releaseIdx = sHeaders.indexOf("ReleaseMode"); //

  const settingRow = sData.find(r => String(r[titleIdx]).trim() === examTitle);
  const releaseMode = settingRow ? String(settingRow[releaseIdx]).toLowerCase() : "hidden";

  // 3. Appliquer la restriction
  if (releaseMode === "hidden") {
    return { error: "Les résultats ne sont pas encore disponibles." };
  }

  // 4. Récupérer le score et les questions (JSON)
  const jsonSh = ss.getSheetByName(EXAM_JSON_SHEET); //
  const jData = jsonSh.getDataRange().getValues();
  let questions = [];
  let score = "";

  for (let i = jData.length - 1; i >= 1; i--) {
    if (String(jData[i][0]).trim() === studentCode && String(jData[i][1]).trim() === examTitle) {
      score = jData[i][3];
      // Si mode manual, on ne renvoie pas les questions, seulement le score
      if (releaseMode === "immediate") {
        questions = JSON.parse(jData[i][4] || "[]");
      }
      break;
    }
  }

  return {
    examTitle: examTitle,
    score: score,
    questions: questions,
    mode: releaseMode
  };
}
function reviewSubmission(updates) {
  const email = Session.getActiveUser().getEmail();
  if (!isAdminEmail(email)) throw new Error("Access denied");

  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  const sh = ss.getSheetByName(SCORES_SHEET);
  if (!sh) throw new Error("Scores sheet not found");

  const data = sh.getDataRange().getValues();
  const header = data[0].map(h => String(h || "").trim());

  const codeIdx = header.indexOf("Student Code");
  if (codeIdx === -1) throw new Error("Student Code column missing");

  // Find exam column (title preferred, fallback to slug)
  let examColIdx = header.indexOf(updates.examTitle);
  if (examColIdx === -1) {
    examColIdx = header.indexOf(updates.examSlug);
  }
  if (examColIdx === -1) throw new Error("Exam column not found");

  // Find student row
  let rowIndex = -1;
  for (let r = 1; r < data.length; r++) {
    if (String(data[r][codeIdx]).trim() === String(updates.studentCode).trim()) {
      rowIndex = r + 1;
      break;
    }
  }
  if (rowIndex === -1) throw new Error("Student not found");

  // 1️⃣ Save final score
  sh.getRange(rowIndex, examColIdx + 1).setValue(updates.finalScore);

  // 2️⃣ Archive correction JSON
  let jsonSh = ss.getSheetByName("Exam_JSON");
  if (!jsonSh) {
    jsonSh = ss.insertSheet("Exam_JSON");
    jsonSh.appendRow([
      "Student Code",
      "Exam Slug",
      "Timestamp",
      "Score",
      "JSON",
      "Corrected_By"
    ]);
  }

  jsonSh.appendRow([
    updates.studentCode,
    updates.examSlug,
    new Date(),
    updates.finalScore,
    JSON.stringify(updates.corrections || []),
    email
  ]);

  // 3️⃣ SEND CORRECTION + SCORE TO STUDENT (FINAL RELEASE)
  submitScoreAndSend({
    studentCode: updates.studentCode,
    studentNameProvided: updates.studentName || "",
    subjectName: updates.examTitle,
    score: updates.finalScore,
    corrections: updates.corrections || []
  });

  return { success: true };
}

function getDriveThumbnail(url) {
  if (!url) return null;

  // Case 1: ?id=FILEID
  let idMatch = url.match(/id=([a-zA-Z0-9_-]+)/);
  if (idMatch) return "https://drive.google.com/thumbnail?id=" + idMatch[1];

  // Case 2: /d/FILEID/
  idMatch = url.match(/\/d\/([a-zA-Z0-9_-]+)\//);
  if (idMatch) return "https://drive.google.com/thumbnail?id=" + idMatch[1];

  // Case 3: direct ID only
  if (/^[a-zA-Z0-9_-]{20,}$/.test(url)) {
    return "https://drive.google.com/thumbnail?id=" + url;
  }

  // Default: return original
  return url;
}

/*************************************************
 * HASH PIN — SHA256
 *************************************************/
function hashPin(pin) {
  const raw = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256,
    pin,
    Utilities.Charset.UTF_8
  );

  return raw.map(b => {
    const v = (b < 0 ? b + 256 : b).toString(16);
    return v.length === 1 ? "0" + v : v;
  }).join("");
}

/**
 * 🚪 PORTIER DE CONNEXION : Authentification Staff
 * Vérifie les identifiants, le statut du compte et les exigences de sécurité.
 */
function attemptSheetLogin(email, password) {
  try {
    const ss = DB.openById(MASTER_SPREADSHEET_ID);
    const sh = ss.getSheetByName("Users");
    if (!sh) return { success: false, message: "Base de données Users introuvable." };

    const data = sh.getDataRange().getValues();
    const headers = data[0].map(h => String(h).trim());

    // 🗺️ MAPPING DYNAMIQUE DES COLONNES
    const idx = {
      email:  headers.indexOf("Email"),
      pass:   headers.indexOf("Password"),
      active: headers.indexOf("Active"),
      reset:  headers.indexOf("Reset_Req")
    };

    // Validation de la configuration
    if (idx.email === -1 || idx.pass === -1 || idx.active === -1) {
      return { success: false, message: "Erreur de configuration : Colonnes manquantes." };
    }

    const target = email.toLowerCase().trim();
    const hashed = hashPin(password);

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      
      // 1️⃣ VÉRIFICATION IDENTIFIANTS (Email & Mot de Passe)
      if (String(row[idx.email]).toLowerCase().trim() === target && String(row[idx.pass]).trim() === hashed) {
        
        // 2️⃣ VÉRIFICATION DU STATUT (Anti-Suspension)
        // On vérifie si la colonne Active contient "TRUE"
        const isActive = String(row[idx.active]).toUpperCase() === "TRUE";
        if (!isActive) {
          return { success: false, message: "🚫 Accès refusé : Votre compte a été suspendu par l'administration." };
        }

        // 3️⃣ VÉRIFICATION DU FORÇAGE DE RÉINITIALISATION
        const needsReset = String(row[idx.reset]).toUpperCase() === "TRUE";
        if (needsReset) {
          return { 
            success: true, 
            status: "RESET_REQUIRED", 
            message: "Première connexion : Veuillez définir votre mot de passe définitif." 
          };
        }

        // 4️⃣ GÉNÉRATION DE SESSION (Succès)
        const token = Utilities.getUuid();
        CacheService.getScriptCache().put(token, target, 21600); // Valide 6 heures

        return { success: true, status: "AUTHORIZED", token: token };
      }
    }

    return { success: false, message: "Identifiants incorrects." };

  } catch (e) {
    console.error("Erreur Login : " + e.message);
    return { success: false, message: "Erreur système lors de la connexion." };
  }
}

function submitScoreAndSend(payload) {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  
  // 1. IDENTIFICATION DE L'ÉTUDIANT (Dynamique)
  const userSh = ss.getSheetByName(MASTER_SHEET_NAME); 
  if (!userSh) {
    Logger.log("ERREUR: Feuille utilisateur '" + MASTER_SHEET_NAME + "' introuvable.");
    return { success: false, error: "Feuille utilisateur introuvable" };
  }

  const userData = userSh.getDataRange().getValues();
  const userHeaders = userData[0].map(h => String(h).trim()); // Nettoyage des en-têtes (espaces cachés)
  
  // Recherche des index des colonnes par leurs noms exacts (Sensible à la casse)
  const colIdIdx = userHeaders.indexOf("Student ID");
  const colNameIdx = userHeaders.indexOf("Student_Name");
  
  // Détection de l'email (vérifie 'Email' ou 'Student Email')
  let colEmailIdx = userHeaders.indexOf("Email");
  if (colEmailIdx === -1) colEmailIdx = userHeaders.indexOf("Student Email");

  const studentRow = userData.find(r => String(r[colIdIdx]).trim() === String(payload.studentCode).trim());
  
  let userEmail = null;
  let studentFullName = payload.studentNameProvided;

  if (studentRow) {
    if (colEmailIdx !== -1) userEmail = String(studentRow[colEmailIdx]).trim();
    if (colNameIdx !== -1) studentFullName = String(studentRow[colNameIdx]).trim();
  }

  Logger.log("DEBUG: Étudiant=" + studentFullName + " | Email trouvé=" + userEmail);

  // 2. GESTION DE LA FEUILLE "SCORES"
  const scoresSh = ss.getSheetByName(SCORES_SHEET);
  let scoresHeaders = scoresSh.getRange(1, 1, 1, scoresSh.getLastColumn()).getValues()[0];
  
  const targetSubject = String(payload.subjectName).trim();
  let colIdx = scoresHeaders.findIndex(h => String(h).trim().toLowerCase() === targetSubject.toLowerCase());

  // CRÉATION AUTOMATIQUE DE L'EXAMEN DANS SCORES SI NON TROUVÉ
  if (colIdx === -1) {
    const newCol = scoresSh.getLastColumn() + 1;
    scoresSh.getRange(1, newCol).setValue(targetSubject)
      .setBackground("#6a1b9a") // Violet Meigens Tech
      .setFontColor("white")
      .setFontWeight("bold");
    
    colIdx = newCol - 1;
  }

  // 3. TROUVER OU CRÉER LA LIGNE ÉTUDIANT DANS "SCORES"
  const scoresData = scoresSh.getDataRange().getValues();
  let rowIdx = -1;
  let existingScoreValue = "";

  for (let i = 1; i < scoresData.length; i++) {
    if (String(scoresData[i][0]).trim() === String(payload.studentCode).trim()) {
      rowIdx = i + 1;
      existingScoreValue = scoresData[i][colIdx];
      break;
    }
  }

  if (rowIdx === -1) {
    scoresSh.appendRow([payload.studentCode, studentFullName]);
    rowIdx = scoresSh.getLastRow();
  }

  // 4. LOGIQUE DE CALCUL ET PLAFONNEMENT (RETAKE)
  let finalScore = payload.score;
  if (existingScoreValue !== "" && existingScoreValue !== null && !isNaN(parseFloat(existingScoreValue)) && finalScore > 70) {
    finalScore = 70; 
  }

  // Sauvegarde de la note
  scoresSh.getRange(rowIdx, colIdx + 1).setValue(finalScore);

  // 5. ARCHIVAGE DANS "EXAM_JSON"
  const jsonSh = ss.getSheetByName(EXAM_JSON_SHEET) || ss.insertSheet(EXAM_JSON_SHEET);
  jsonSh.appendRow([
    payload.studentCode,
    targetSubject,
    new Date(),
    finalScore,
    JSON.stringify(payload.corrections),
    "", // Teacher Comments
    "TRUE" // Released
  ]);

  // 6. ENVOI DE L'EMAIL (Validation stricte)
  if (userEmail && userEmail.includes("@")) {
    try {
      sendCorrectionEmail(userEmail, targetSubject, studentFullName, finalScore, payload.corrections);
      Logger.log("SUCCÈS: Email de correction envoyé à " + userEmail);
    } catch (e) {
      Logger.log("ERREUR: Échec de l'envoi de l'email : " + e.message);
    }
  } else {
    Logger.log("AVERTISSEMENT: Email invalide ou manquant dans la feuille Student_Users. Envoi annulé.");
  }

  return { success: true, finalScore: finalScore };
}

function updateExamSettings(payload) {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  const sh = ss.getSheetByName(SETTINGS_SHEET);
  const data = sh.getDataRange().getValues();
  const headers = data[0];
  
  // 1. Détection dynamique des colonnes pour éviter tout décalage futur
  const titleIdx = headers.indexOf("ExamTitle");     // Position de 'ExamTitle'
  const acceptIdx = headers.indexOf("AcceptResponse") !== -1 ? headers.indexOf("AcceptResponse") : headers.indexOf("AcceptResponses");
  const dueIdx = headers.indexOf("DueDate");
  const releaseIdx = headers.indexOf("ReleaseMode");
  const typeIdx = headers.indexOf("ExamType");

  const targetTitle = String(payload.ExamTitle).trim();
  let rowIndex = -1;

  // 2. Recherche de la ligne par le titre de l'examen
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][titleIdx]).trim() === targetTitle) {
      rowIndex = i + 1;
      break;
    }
  }

  // 3. Mise à jour ou Création
  if (rowIndex !== -1) {
    // On met à jour chaque colonne trouvée dynamiquement
    if (acceptIdx !== -1) sh.getRange(rowIndex, acceptIdx + 1).setValue(payload.AcceptResponses);
    if (dueIdx !== -1) sh.getRange(rowIndex, dueIdx + 1).setValue(payload.DueDate);
    if (releaseIdx !== -1) sh.getRange(rowIndex, releaseIdx + 1).setValue(payload.ReleaseMode);
    if (typeIdx !== -1) sh.getRange(rowIndex, typeIdx + 1).setValue(payload.ExamType);
    
    Logger.log("Mise à jour réussie pour : " + targetTitle);
    return { success: true };
  } else {
    // Si l'examen n'existe pas dans Settings, on ajoute une nouvelle ligne
    // Ordre : Settings, Value, OneSub, Accept, Active, Admins, DueDate, Release, Title, Type
    // Note: Ajustez l'ordre ci-dessous si vous voulez un appendRow précis
    sh.appendRow([
      "Autorité", "TRUE", "TRUE", 
      payload.AcceptResponses, 
      "Exam", "", 
      payload.DueDate, 
      payload.ReleaseMode, 
      targetTitle, 
      payload.ExamType
    ]);
    return { success: true, status: "created" };
  }
}

/**
 * CRITICAL: Ensures all required sheets exist and have the correct headers.
 * Automatically repairs missing columns (like Password/Reset_Req) without data loss.
 */
function checkAndInitSheets() {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  
  // Define the exact structure we need
  const sheetDefinitions = [
    {
      name: "Users",
      // 10 Columns total for the new Auth system
      headers: [
        "Email",       // A
        "Role",        // B
        "p_pic",       // C
        "p_name",      // D
        "p_dossier",   // E
        "p_settings",  // F
        "p_manual",    // G
        "p_review",    // H
        "Password",    // I (Hashed)
        "Reset_Req"    // J (TRUE/FALSE)
      ],
      color: "#d1d1d1" // Grey
    },
    {
      name: "Settings",
      headers: ["Settings", "Value", "OneSubmission", "AcceptResponses", "ExamActive", "Admins", "DueDate", "ReleaseMode", "ExamTitle", "ExamType", "Whitelist", "ThemeJSON"],
      color: "#f3f3f3"
    },
    {
      name: "Scores",
      headers: ["Student Code", "Student Name"], // Only enforce the first 2, others are dynamic exams
      color: "#e8f5e9"
    },
    {
      name: "Exam_JSON",
      headers: ["Student Code", "Exam Slug", "Timestamp", "Score", "JSON", "Corrected_By", "Released"],
      color: "#fff3e0"
    },
    {
      name: "Exam_Questions",
      headers: ["ExamTitle", "QuestionText", "Option1", "Option2", "Option3", "Option4", "CorrectIndex", "Commentary", "MediaType", "MediaURL", "ResponseType"],
      color: "#e3f2fd"
    }
  ];

  sheetDefinitions.forEach(def => {
    let sheet = ss.getSheetByName(def.name);
    
    // --- SCENARIO 1: SHEET DOES NOT EXIST (Create it) ---
    if (!sheet) {
      sheet = ss.insertSheet(def.name);
      
      // Write Headers
      sheet.getRange(1, 1, 1, def.headers.length)
           .setValues([def.headers])
           .setFontWeight("bold")
           .setBackground(def.color);
      
      sheet.setFrozenRows(1);
      Logger.log(`Created missing sheet: ${def.name}`);

      // Special Case: Auto-add Owner to 'Users' sheet so you aren't locked out
      if (def.name === "Users") {
        const ownerEmail = Session.getEffectiveUser().getEmail();
        sheet.appendRow([
          ownerEmail, "Master", "TRUE", "TRUE", "TRUE", "TRUE", "TRUE", "TRUE", "GOOGLE_AUTH", "FALSE"
        ]);
      }
    } 
    
    // --- SCENARIO 2: SHEET EXISTS (Check for missing columns) ---
    else {
      const currentLastCol = sheet.getLastColumn();
      
      // If the sheet is empty (no headers at all), just write them
      if (currentLastCol === 0) {
         sheet.getRange(1, 1, 1, def.headers.length)
           .setValues([def.headers])
           .setFontWeight("bold")
           .setBackground(def.color);
         return;
      }

      // Read existing headers
      const currentHeaders = sheet.getRange(1, 1, 1, currentLastCol).getValues()[0].map(h => String(h).trim());

      // Loop through required headers and check if they exist
      def.headers.forEach((reqHeader, index) => {
        // We use the index to check strictly (Col 1 should be Email, Col 9 Password, etc.)
        // Or we check if it exists *anywhere* if order doesn't matter (usually order matters for mapping)
        
        // Simple approach: If the sheet has fewer columns than required, append the missing ones at the end
        if (index >= currentLastCol) {
           // This header is missing (e.g., 'Password' is at index 8, but sheet only has 8 columns (0-7))
           sheet.getRange(1, index + 1).setValue(reqHeader)
                .setFontWeight("bold")
                .setBackground(def.color);
           Logger.log(`Repaired ${def.name}: Added missing header '${reqHeader}'`);
        }
        else {
           // Optional: Fix header name if it's wrong/renamed? 
           // Usually better to leave existing data alone unless empty.
           const existingHeader = currentHeaders[index];
           if (!existingHeader) {
              sheet.getRange(1, index + 1).setValue(reqHeader).setFontWeight("bold");
           }
        }
      });
    }
  });
}

/**
 * Fetches questions from "Exam_Questions" and applies the limit 
 * based on ExamType found in "Settings".
 */
function getQuestionsForStudent(examTitle) {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  
  // 1. Get the Limit from Settings
  const settingsSh = ss.getSheetByName(SETTINGS_SHEET);
  const settingsData = settingsSh.getDataRange().getValues();
  const settingsHeaders = settingsData[0];
  const typeIdx = settingsHeaders.indexOf("ExamType"); // Col J
  const titleIdx = settingsHeaders.indexOf("ExamTitle"); // Col G
  
  let limit = 10; // Default fallback
  const settingRow = settingsData.find(r => String(r[titleIdx]).trim() === String(examTitle).trim());
  
  if (settingRow) {
    const type = String(settingRow[typeIdx]).toLowerCase().trim();
    if (type === "homework") {
      limit = 30;
    } else {
      limit = 10; // "examen"
    }
  }

  // 2. Get All Questions for this Subject
  const qSh = ss.getSheetByName("Exam_Questions");
  const qData = qSh.getDataRange().getValues();
  qData.shift(); // Remove headers

  let filteredQuestions = qData
    .filter(r => String(r[0]).trim() === String(examTitle).trim())
    .map(r => ({
      text: r[1],
      options: [r[2], r[3], r[4], r[5]].filter(o => o !== ""),
      answer: r[6],
      mediaType: r[8],
      mediaUrl: r[9],
      type: r[10]
    }));

  // 3. Shuffle and Slice to the limit
  return shuffleArray(filteredQuestions).slice(0, limit);
}

// Helper to shuffle questions so every student gets a different order
function shuffleArray(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }
  return array;
}
/**
 * GET EXAM SETTINGS (Prevents Null Crash)
 */
function getExamSettings(examTitle = null) {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  let sh = ss.getSheetByName("Settings");
  
  // Default Safe Object
  const defaults = {
    AcceptResponses: "FALSE",
    ReleaseMode: "hidden",
    ExamType: "examen",
    DueDate: ""
  };

  // If sheet doesn't exist, create it with correct headers
  if (!sh) {
    sh = ss.insertSheet("Settings");
    const headers = ["ExamTitle", "AcceptResponses", "ReleaseMode", "ExamType", "DueDate", "OneSubmission", "ExamActive", "ThemeJSON"];
    sh.getRange(1, 1, 1, headers.length).setValues([headers]).setFontWeight("bold");
    return defaults;
  }

  const data = sh.getDataRange().getValues();
  // Safe Header Lookup
  const h = data[0].map(x => String(x).toLowerCase().trim());
  const titleIdx = h.indexOf("examtitle");

  // If "ExamTitle" column is missing, return defaults
  if (titleIdx === -1) return defaults;

  // Search for the specific exam
  for (let i = 1; i < data.length; i++) {
    const rowTitle = String(data[i][titleIdx]).trim();
    // Return specific exam settings OR the first one found if none specified
    if ((examTitle && rowTitle === examTitle) || (!examTitle && rowTitle !== "")) {
      return {
        AcceptResponses: String(data[i][h.indexOf("acceptresponses")]).toLowerCase() === "true" ? "true" : "false",
        ReleaseMode: data[i][h.indexOf("releasemode")] || "hidden",
        ExamType: data[i][h.indexOf("examtype")] || "examen",
        DueDate: data[i][h.indexOf("duedate")] || ""
      };
    }
  }
  
  return defaults;
}

/**
 * Recherche les étudiants basés sur le statut ReleaseMode de la feuille Settings
 */
function getUnreleasedStudents(examTitle) {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID); //
  const settingsSh = ss.getSheetByName(SETTINGS_SHEET); //
  const sData = settingsSh.getDataRange().getValues();
  const sHeaders = sData[0];
  
  const titleIdx = sHeaders.indexOf("ExamTitle"); //
  const releaseIdx = sHeaders.indexOf("ReleaseMode"); //
  
  const examSetting = sData.find(r => String(r[titleIdx]).trim() === String(examTitle).trim());
  
  // Si déjà en mode 'immediate' dans Settings, on considère tout libéré
  if (examSetting && String(examSetting[releaseIdx]).toLowerCase() === "immediate") {
    return [];
  }

  const scoresSh = ss.getSheetByName(SCORES_SHEET); //
  const data = scoresSh.getDataRange().getValues();
  const headers = data[0];
  const examColIdx = headers.indexOf(examTitle);
  const nameIdx = headers.indexOf("Student Name");

  const unreleasedList = [];
  for (let i = 1; i < data.length; i++) {
    const score = data[i][examColIdx];
    if (score !== "" && score !== null) {
      unreleasedList.push({
        id: data[i][0], 
        name: data[i][nameIdx] || "Étudiant " + data[i][0],
        score: score
      });
    }
  }
  return unreleasedList;
}

/**
 * Envoie les emails et met à jour ReleaseMode dans Settings vers 'immediate'
 */
function processBulkRelease(studentIds, examTitle) {
  let sentCount = 0;
  studentIds.forEach(id => {
    try {
      emailDetailedCorrection(id, examTitle); //
      sentCount++;
    } catch (e) {
      Logger.log("Erreur d'envoi pour l'ID " + id + ": " + e.message);
    }
  });

  // Mise à jour de la feuille Settings
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID); //
  const sh = ss.getSheetByName(SETTINGS_SHEET); //
  const data = sh.getDataRange().getValues();
  const headers = data[0];
  const titleIdx = headers.indexOf("ExamTitle"); //
  const releaseIdx = headers.indexOf("ReleaseMode"); //

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][titleIdx]).trim() === String(examTitle).trim()) {
      sh.getRange(i + 1, releaseIdx + 1).setValue("immediate"); //
      break;
    }
  }

  return { sent: sentCount };
}
/**
 * Automatically syncs data from Source to Archive on load.
 * Includes a 10-minute cooldown to prevent performance issues.
 */
function syncStudentDataOnLoad() {
  const cache = CacheService.getScriptCache();
  const cached = cache.get("DATA_SYNCED_RECENTLY");

  // If we synced in the last 10 minutes, skip to save speed
  if (cached) {
    Logger.log("Skipping sync: Data synced recently.");
    return; 
  }

  Logger.log("Starting Auto-Archive Sync...");

  // 1. Setup Spreadsheets
  const sourceSS = SpreadsheetApp.openById(SOURCE_SPREADSHEET_ID);
  const sourceSheet = sourceSS.getSheetByName("StudentData");
  
  const masterSS = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  const targetSheet = masterSS.getSheetByName(MASTER_SHEET_NAME); // "Student_Users"

  if (!sourceSheet || !targetSheet) return;

  // 2. Get Data
  const sourceData = sourceSheet.getDataRange().getValues();
  const sourceHeaders = sourceData.shift(); // Remove headers
  
  const targetData = targetSheet.getDataRange().getValues();
  const targetHeaders = targetData.shift(); // Remove headers

  // 3. Map Headers (Find column indexes)
  // Source Map
  const sMap = {
    code: sourceHeaders.indexOf("Code"),
    name: sourceHeaders.indexOf("Nom"),
    phone: sourceHeaders.indexOf("Téléphone"),
    address: sourceHeaders.indexOf("Adresse"),
    photo: sourceHeaders.indexOf("PhotoURL"),
    blood: sourceHeaders.indexOf("Groupe Sanguin")
  };

  // Target Map (Archive)
  const tMap = {
    id: targetHeaders.indexOf("Student ID"),
    name: targetHeaders.indexOf("Student_Name"),
    phone: targetHeaders.indexOf("Phone_Number"),
    address: targetHeaders.indexOf("Address"),
    photo: targetHeaders.indexOf("Picture_URL"),
    blood: targetHeaders.indexOf("Blood_Type")
  };

  if (sMap.code === -1 || tMap.id === -1) return; // Cannot map IDs

  // 4. Create Map of Existing Archive Users for fast lookup
  const archiveMap = new Map();
  targetData.forEach((row, index) => {
    const id = String(row[tMap.id]).trim().toUpperCase();
    if (id) archiveMap.set(id, index + 2); // Store Row Number (1-based index + header)
  });

  // 5. Iterate Source and Update/Append
  let updates = 0;
  let newRecords = 0;

  sourceData.forEach(row => {
    const code = String(row[sMap.code]).trim().toUpperCase();
    if (!code) return;

    // Helper to get source value safely
    const getVal = (idx) => (idx !== -1 && row[idx]) ? row[idx] : "";

    if (archiveMap.has(code)) {
      // --- UPDATE EXISTING ---
      const rowNum = archiveMap.get(code);
      // Only update if source has data (Don't overwrite with empty)
      if (getVal(sMap.name)) targetSheet.getRange(rowNum, tMap.name + 1).setValue(getVal(sMap.name));
      if (getVal(sMap.phone)) targetSheet.getRange(rowNum, tMap.phone + 1).setValue(getVal(sMap.phone));
      if (getVal(sMap.photo)) targetSheet.getRange(rowNum, tMap.photo + 1).setValue(getVal(sMap.photo));
      if (getVal(sMap.address)) targetSheet.getRange(rowNum, tMap.address + 1).setValue(getVal(sMap.address));
      if (getVal(sMap.blood)) targetSheet.getRange(rowNum, tMap.blood + 1).setValue(getVal(sMap.blood));
      updates++;
    } else {
      // --- APPEND NEW ---
      // Create an empty row with correct length
      const newRow = new Array(targetHeaders.length).fill("");
      
      // Fill known fields
      newRow[tMap.id] = code;
      newRow[tMap.name] = getVal(sMap.name);
      newRow[tMap.phone] = getVal(sMap.phone);
      newRow[tMap.photo] = getVal(sMap.photo);
      newRow[tMap.address] = getVal(sMap.address);
      newRow[tMap.blood] = getVal(sMap.blood);
      
      targetSheet.appendRow(newRow);
      newRecords++;
    }
  });

  Logger.log(`Sync Complete. Updated: ${updates}, New: ${newRecords}`);
  
  // Set Cache to prevent running again for 10 minutes (600 seconds)
  cache.put("DATA_SYNCED_RECENTLY", "true", 600);
}

/**
 * 2. GETS CURRENT USER (Supports Google ID OR Token)
 */
function getEffectiveAdminUser(token) {
  // A. Try Google Login (Admin Link)
  const googleEmail = Session.getActiveUser().getEmail();
  if (googleEmail) return googleEmail;

  // B. Try Token Login (Student Link Fallback)
  if (token) {
    const cachedEmail = CacheService.getScriptCache().get(token);
    if (cachedEmail) return cachedEmail;
  }

  return null; // Not logged in
}

function verifyStudentByLast4(last4) {
  const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
  const sh = ss.getSheetByName(MASTER_SHEET_NAME); 
  const data = sh.getDataRange().getValues();
  const headers = data[0];
  
  // Robust Column Lookup
  const idIdx = headers.indexOf("Student ID");
  const nameIdx = headers.indexOf("Student_Name");
  
  // This looks for "Picture_URL", "PhotoURL", or "Photo" automatically
  let picIdx = headers.indexOf("Picture_URL");
  if (picIdx === -1) picIdx = headers.indexOf("PhotoURL");
  if (picIdx === -1) picIdx = headers.indexOf("Photo");

  if (idIdx === -1) return { found: false, error: "ID column not found" };

  const target = String(last4).trim();
  
  for (let i = 1; i < data.length; i++) {
    const fullId = String(data[i][idIdx]).trim();
    
    if (fullId.endsWith(target) && target.length >= 4) {
       let rawUrl = (picIdx !== -1) ? String(data[i][picIdx]).trim() : "";
       let cleanUrl = "";

       // Convert Google Drive links to thumbnails
       if (rawUrl && rawUrl.includes("drive.google.com")) {
         let idMatch = rawUrl.match(/id=([a-zA-Z0-9_-]+)/) || rawUrl.match(/\/d\/([a-zA-Z0-9_-]+)\//);
         if (idMatch) cleanUrl = "https://drive.google.com/thumbnail?id=" + idMatch[1] + "&sz=w300";
       } else {
         cleanUrl = rawUrl; // Use as-is if not Drive link
       }

       return {
         found: true,
         student: {
           Student_ID: fullId,
           Student_Name: (nameIdx !== -1 ? data[i][nameIdx] : "Étudiant"),
           Picture_URL: cleanUrl
         }
       };
    }
  }
  return { found: false };
}
/**
 * Vérifie si une note existe déjà pour un étudiant et un examen précis.
 */
function getExistingGrade(studentCode, examTitle) {
  try {
    const ss = SpreadsheetApp.openById(SUBMISSION_SHEET_ID);
    const sh = ss.getSheetByName(SCORES_SHEET);
    
    if (!sh) return { found: false };

    const data = sh.getDataRange().getValues();
    const headers = data[0].map(h => String(h).trim());

    const codeIdx = headers.indexOf("Student Code");
    const examColIdx = headers.indexOf(String(examTitle).trim());

    if (codeIdx === -1 || examColIdx === -1) return { found: false };

    // Recherche de l'étudiant par son code
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][codeIdx]).trim() === String(studentCode).trim()) {
        const grade = data[i][examColIdx];
        // On considère que la note existe si la cellule n'est pas vide
        if (grade !== "" && grade !== null) {
          return { found: true, grade: grade };
        }
        break; 
      }
    }
    return { found: false };

  } catch (e) {
    Logger.log("Erreur dans getExistingGrade: " + e.message);
    return { found: false, error: e.message };
  }
}
